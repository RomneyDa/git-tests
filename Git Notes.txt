// GIT NOTES

$ git init       // initializes a local git repository
$ git add <file> // Adds file(s) to index; for example, git add *.html adds all html files, git add . adds all files
$ git rm <file>  // Removes file(s) from index
$ git status     // checks the status of the working tree
$ git commit     // commit changes index to LOCAL repository
$ git push       // pushes to remote repository
$ git pull       // Pulls the latest version from remote repository
$ git clone      // Copies a remote repository into your current folder

$ git config --global user.name '<name>'  // set user name
$ git config --global user.email '<email' // set user email
$ git status // checks status of git in current folder
